let rec parse_str = fun text level ->
  if eq_string text "" then [], ""
  else if nth_char text 0 = `\n` then parse_str (sub_string text 1 (string_length text - 1)) level
  else (
	match (sub_string text 0 2) with
	| "Ba" -> let p, next = parse_str (sub_string text 2 (string_length text - 2)) level in (Ba :: p), next
	| "tt" -> let p, next = parse_str (sub_string text 2 (string_length text - 2)) level in (tt :: p), next
	| "le" -> let p, next = parse_str (sub_string text 2 (string_length text - 2)) level in (le :: p), next
	| "co" -> let p, next = parse_str (sub_string text 2 (string_length text - 2)) level in (co :: p), next
	| "de" -> let p, next = parse_str (sub_string text 2 (string_length text - 2)) level in (de :: p), next
	| "ro" -> let p, next = parse_str (sub_string text 2 (string_length text - 2)) (level+1) in
						      let p2, next2 = parse_str next level in (ro p :: p2), next2
	| "ck" -> [], (sub_string text 3 (string_length text - 3))
  );;
  
  
let rec word = fun l ->
 match l with
  | [] -> 0
  | (ro t)::q -> word t + word q + 5
  | t::q -> word q + 2;;
  
  
let run_str = fun text ->
  let file = open_out_bin "D:\\Profiles\\dBarraud\\Documents\\BattleCode\\result.txt" in
  let charcount = ref 0 in
  let codeparsed,_ = parse_str text 0 in
  let cursor = ref 0 in
  let vector = ref (make_vect 1000 0) in
  let rec run_code = fun code ->
  match code with
   | [] -> ()
   | todo :: rest ->
   ( 
  ( match todo with
  | Ba -> ( cursor := !cursor + 1; if vect_length !vector = !cursor then vector := concat_vect !vector (make_vect 1000 0))
  | tt -> if !cursor > 0 then cursor := !cursor - 1
  | le -> !vector.(!cursor) <- !vector.(!cursor) + 1
  | co -> !vector.(!cursor) <- !vector.(!cursor) - 1
  | de -> ( output_char file (char_of_int (!vector.(!cursor))); if !charcount > 10 then (flush file; charcount := 0) else (charcount := !charcount + 1) )
  | ro code2 ->  while (!vector.(!cursor) > 0) do run_code code2 done )
  ;
  run_code rest )
in
let t = time () in
run_code codeparsed;
let t_total = int_of_float (time () -. t) in
output_char file `\n`;
output_string file (string_of_int t_total);
output_char file `\n`;
close_out file;;
